Using spark (demo)

Identified the most “popular” users on the Twitter network, measured by the number of their followers.

dataset: twitter.edges


    // topUser.scala
    val file = "twitter.edges"
    val lines = sc.textFile(file)
    val splitRdd = lines.map(line => line.split(": "))
    val yourRdd = splitRdd.flatMap( arr => {
      val follower = arr( 0 )
      val manySubject = arr( 1 )
      val subjects = manySubject.split(",")
      subjects.map( subject => ( subject, 1 ) )
    } )
    val Counts = yourRdd.reduceByKey((a,b) => a+b)
    val finalCount = Counts.filter({case (key,value) => value > 1000})

    finalCount.saveAsTextFile("output")
    System.exit(0)
